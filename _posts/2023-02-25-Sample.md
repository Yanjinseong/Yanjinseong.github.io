```c++
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
	int n;
	int cnt = 0;
	long sum;

	cin >> n;
	vector<long> v(n, 0);

	for (int i = 0; i < n; i++)
	{
		cin >> v[i];
	}

	sort(v.begin(), v.end());

	for (int k = 0; k < n; k++)
	{
		long find = v[k];
		int i = 0;
		int j = n - 1;

		while (i < j) // 투 포인터 알고리즘 사용
		{
			sum = v[i] + v[j];
			if (sum < find) 
				i++;
			else if (sum > find)
				j--;
			else
			{
				if (i != k && j != k) // 찾는 수의 합이 산출된 경우
				{
					cnt++;
					break;
				}
				else if (i == k)
					i++;
				else if (j == k)
					j--;
			}
		}
	}
	cout << cnt << '\n';
	return 0;
}
```
